# In production Debug mode should be off

app = Flask(__name__)
# Flask is the class, app is the object, __name__ refers to python script from where we are working

@app.route('/about/<username>')
def about_page(username):
    return f'<h1>This is the about page of {username}'
# @app indicates the app Decorator
# route indicates the path of function
# <username> allows dynamic routes where webpage can happen based on user name instead static

# If we make template folder inside the folder (where flask file is there), then automatically undestood will save html files there

@app.route('/')
@app.route('/home')
def home_page():
    return render_template('home.html')  
# render_template will import from flask
